```python
df_guests['age'] = pd.to_datetime(df_guests['date_of_birth']).apply(

)
```

## KYC ANOMALY SIGNALS SUMMARY


```python
# Consolidate KYC signals
df_kyc_full = df_bookings.merge(
    df_guests[['guest_id', 'nationality', 'mobility_impaired', 'date_of_birth']], 
    on='guest_id'
).merge(
    df_stays_beha[['booking_id', 'refused_additional_info', 'num_complaints', 
              'incident_registered', 'overall_satisfaction']],
    on='booking_id', how='left'
)

# KYC Flags
df_kyc_full['flag_payment_mismatch']    = (df_kyc_full['payment_country'] != df_kyc_full['nationality']).astype(int)
df_kyc_full['flag_last_minute']         = (df_kyc_full['advance_booking_days'] <= 3).astype(int)
df_kyc_full['flag_high_modifications']  = (df_kyc_full['num_modifications'] >= 3).astype(int)
df_kyc_full['flag_refused_info']        = df_kyc_full['refused_additional_info'].fillna(0).astype(int)
df_kyc_full['flag_no_show']             = (df_kyc_full['booking_status'] == 'No Show').astype(int)

kyc_flags = ['flag_payment_mismatch', 'flag_last_minute', 
             'flag_high_modifications', 'flag_refused_info', 'flag_no_show']

kyc_rates = df_kyc_full[kyc_flags].mean() * 100

fig, ax = plt.subplots(figsize=(12, 5))
kyc_rates.plot(kind='bar', ax=ax, color='crimson', edgecolor='white')
ax.set_title('KYC Anomaly Signal Rates (%)')
ax.set_ylabel('% of Bookings')
ax.set_xticklabels([
    'Payment\nMismatch', 'Last Minute\nBooking',
    'High\nModifications', 'Refused\nInfo', 'No Show'
], rotation=0)
plt.tight_layout()
plt.savefig("..\Images\EDA\KYC_Anomalies\Anomaly_summary.png")
plt.show()

print('\nKYC Signal Rates:')
for flag, rate in kyc_rates.items():
```
